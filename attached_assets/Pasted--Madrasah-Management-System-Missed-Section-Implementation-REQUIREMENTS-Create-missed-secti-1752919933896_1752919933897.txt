-- Timetable Table
CREATE TABLE timetable (
   id INT PRIMARY KEY,
   class_id INT,
   subject_id INT, 
   teacher_id INT,
   day_of_week VARCHAR(10),
   period INT,
   start_time TIME,

);

-- Attendance Records Table  
CREATE TABLE attendance_records (
   id INT PRIMARY KEY,
   class_id INT,
   subject_id INT,
   teacher_id INT,
   period INT,
   attendance_date DATETIME,
   student_id INT,
   status ENUM('present', 'absent')
);
üéØ CORE LOGIC TO IMPLEMENT
Main Query Logic:
sql-- Find classes where scheduled time has passed but no attendance taken
SELECT 
    t.class_id,
    t.subject_id,
    t.teacher_id,
    t.period,
    t.scheduled_date as missed_date,
    t.start_time,
    t.end_time,
    c.class_name,
    s.subject_name,
    te.teacher_name
FROM timetable t
LEFT JOIN attendance_records a ON (
    t.class_id = a.class_id 
    AND t.subject_id = a.subject_id
    AND DATE(a.attendance_date) = t.scheduled_date
    AND a.period = t.period
)
LEFT JOIN classes c ON t.class_id = c.id
LEFT JOIN subjects s ON t.subject_id = s.id  
LEFT JOIN teachers te ON t.teacher_id = te.id
WHERE a.id IS NULL  -- No attendance record exists
AND DATETIME(t.scheduled_date, t.end_time) < NOW()  -- Class time has passed
ORDER BY t.scheduled_date DESC, t.period;
